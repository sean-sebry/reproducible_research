stats_for_steps_by_day <- data_tidy %>% 
  filter(steps > 0) %>% 
  group_by(date) %>% 
  summarize(avg_steps_per_day = mean(steps), median_steps_per_day = as.integer(median(steps))) %>% 
  print()
